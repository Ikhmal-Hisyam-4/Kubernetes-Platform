! ============================================
! S3900-24T4S VLAN Configuration Reference
! ============================================

! Step 1 - Enter config mode
configure

! Step 2 - Create VLANs
vlan 10
name Management
vlan 30
name VM-Network
vlan 40
name Storage
vlan 50
name Kubernetes

! Step 3 - Port 1 (pfSense trunk - all VLANs)
interface ethernet 1/1
switchport mode trunk
switchport trunk allowed vlan 10,30,40,50
switchport trunk native vlan 10

! Step 4 - Port 2 (lab-rancher - management only)
interface ethernet 1/2
switchport mode access
switchport access vlan 10

! Step 5 - Port 3 (lab-hvst-01 NIC1)
interface ethernet 1/3
switchport mode trunk
switchport trunk allowed vlan 10,30,50
switchport trunk native vlan 10

! Step 6 - Port 4 (lab-hvst-01 NIC2 - storage)
interface ethernet 1/4
switchport mode access
switchport access vlan 40

! Step 7 - Port 5 (lab-hvst-02 NIC1)
interface ethernet 1/5
switchport mode trunk
switchport trunk allowed vlan 10,30,50
switchport trunk native vlan 10

! Step 8 - Port 6 (lab-hvst-02 NIC2 - storage)
interface ethernet 1/6
switchport mode access
switchport access vlan 40

! Step 9 - Port 7 (lab-hvst-03 NIC1)
interface ethernet 1/7
switchport mode trunk
switchport trunk allowed vlan 10,30,50
switchport trunk native vlan 10

! Step 10 - Port 8 (lab-hvst-03 NIC2 - storage)
interface ethernet 1/8
switchport mode access
switchport access vlan 40

! Step 11 - Save
end
copy running-config startup-config
